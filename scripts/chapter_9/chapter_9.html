9.5 Statistical transformations

ggplot(diamonds, aes(x = cut)) +
  geom_bar()
# change stat of geom_bar() from count to identity
diamonds |> 
  count(cut) |> 
  ggplot(aes(x = cut, y = n)) +
  geom_bar(stat = "identity")
# display bar chart of proportions
ggplot(diamonds, aes(x = cut, y = after_stat(prop), group = 1)) +
  geom_bar()
ggplot(diamonds) +
  stat_summary(
    aes(x = cut, y = depth),
    fun.min = min,
    fun.max = max,
    fun = median
  )
