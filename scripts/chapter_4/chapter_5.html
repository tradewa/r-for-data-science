5.3.2 How does pivoting work?

# create dataset from scratch using tribble
df <- tribble(
  ~ id, ~bp1, ~bp2,
  "A", 100, 120,
  "B", 140, 115,
  "C", 120, 125
)

# pivot longer 
df |> 
  pivot_longer(
    cols = bp1:bp2,
    names_to = "measurement",
    values_to = "value"
  )
